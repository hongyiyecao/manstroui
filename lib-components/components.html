    生日

                            
    使用方法
    
                        
    直接写id="birthday"的空div
    
                        <div id="birthday"></div>
                         
    加上下面js并引入bday-picker.js和public.js
    
                        
    后续会将css和js进行封装，到时直接引入ManstroUI库文件即可了。
    
                    
                                    用到的JS
                                $(function() {
                                    var datenow = new Date().Format("yyyy-MM-dd");
                                    //用户生日
                                    var birthday = '1991-04-14';
                                    if (birthday != '') {
                                        datenow = birthday;
                                    }
                                    $("#birthday").birthdaypicker({
                                        dateFormat: "bigEndian",
                                        monthFormat: "short",
                                        placeholder: true,
                                        hiddenDate: true,
                                        // 设定默认日期
                                        defaultDate: datenow
                                    });
                                });
