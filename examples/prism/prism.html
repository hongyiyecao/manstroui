$(function() {
        var datenow = new Date().Format("yyyy-MM-dd");
        //用户生日
        var birthday = '1991-04-14';
        if (birthday != '') {
            datenow = birthday;
        }
        $("#birthday").birthdaypicker({
            dateFormat: "bigEndian",
            monthFormat: "short",
            placeholder: true,
            hiddenDate: true,
            // 设定默认日期
            defaultDate: datenow
        });
    });
