用到的js

                $(".nav-tabs li").click(function() {
                    var tabLabel = $(this).attr("id");
                    $(".nav-tabs li").removeClass("active");
                    $(this).addClass("active");
                    $(".tab-content .tab-pane").addClass("hidden");
                    $("." + tabLabel).removeClass('hidden');
                });
